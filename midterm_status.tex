% Slide 2: Project Overview (Review)
\begin{frame}{Project Overview (Review)}
\textbf{Problem Statement}
\begin{itemize}
    \item Current 5G positioning systems process location data in plaintext through centralized Location Management Functions (LMF), enabling potential mass surveillance without authorization controls or privacy protections
\end{itemize}

\vspace{0.5cm}
\textbf{Key Objectives}
\begin{itemize}
    \item Design cryptographic multi-party authorization framework for 5G positioning
    \item Implement privacy-preserving localization using homomorphic encryption
    \item Develop secure network slicing for authorized positioning services
    \item Integrate judicial oversight mechanisms with threshold cryptography
\end{itemize}

\vspace{0.3cm}
\textbf{Success Metrics}
\begin{itemize}
    \item Positioning Accuracy: Maintain $\leq$ 3m with security controls
    \item Authorization Latency: $<$ 5 minutes for judicial approval
\end{itemize}
\end{frame}

\begin{itemize}
    \item \textbf{5G Simulation:} OpenAirInterface core (AMF, SMF, UPF) with custom secure LMF

\begin{itemize}
    \item \textbf{Adversary:} Rogue government agencies or compromised network operators

    \item Judicial authorities maintain secure key management systems
\end{itemize}

% Slide 11: Shamir's Secret Sharing - Concept
\begin{frame}{Shamir's Secret Sharing for Multi-Party Authorization}
\textbf{Problem with Current System}
\begin{itemize}
    \item Single entity (AMF operator/admin) can authorize location requests
    \item No checks and balances $\rightarrow$ Potential for abuse
    \item Demonstrated vulnerability: Anyone with AMF access can track UEs
\end{itemize}

\vspace{0.5cm}
\textbf{Proposed Solution: Threshold Cryptography}
\begin{itemize}
    \item Require multiple independent parties to approve location requests
    \item Use \textbf{(t, n)-Threshold Scheme:} $t$ out of $n$ parties must agree
    \item Proposed: \textbf{(3, 5) scheme} - 3 out of 5 parties must approve
\end{itemize}

\vspace{0.5cm}
\textbf{5 Authorization Parties:}
\begin{enumerate}
    \item \textbf{Judicial Authority} - Court order validation
    \item \textbf{Law Enforcement Agency} - Investigation justification
    \item \textbf{Network Operator Security Officer} - Technical feasibility
    \item \textbf{Privacy Oversight Officer} - Privacy impact assessment
    \item \textbf{Independent Auditor} - Compliance verification
\end{enumerate}

\vspace{0.3cm}
\textbf{Key Property:} Cannot track UE unless 3+ parties collude
\end{frame}

% Slide 12: Shamir's Secret Sharing - Mathematical Foundation
\begin{frame}{Shamir's Secret Sharing - Mathematical Foundation}
\textbf{How It Works (Shamir 1979):}

\vspace{0.3cm}
\textbf{1. Secret Generation \& Sharing}
\begin{itemize}
    \item Authorization secret $S$ (e.g., LMF access key)
    \item Encode as polynomial of degree $t-1$:
    $$f(x) = S + a_1x + a_2x^2 + \ldots + a_{t-1}x^{t-1} \pmod{p}$$
    where $p$ is large prime, $a_i$ are random coefficients
    \item Each party $i$ receives share: $(i, f(i))$
\end{itemize}

\vspace{0.3cm}
\textbf{2. Secret Reconstruction}
\begin{itemize}
    \item Collect $t$ shares: $(x_1, y_1), (x_2, y_2), \ldots, (x_t, y_t)$
    \item Use Lagrange interpolation to reconstruct polynomial:
    $$S = f(0) = \sum_{j=1}^{t} y_j \prod_{k=1, k \neq j}^{t} \frac{x_k}{x_k - x_j}$$
    \item Recovered $S$ authorizes location request to LMF
\end{itemize}

\vspace{0.3cm}
\textbf{Security Properties:}
\begin{itemize}
    \item \textbf{Perfect Security:} Fewer than $t$ shares reveal \textit{zero} information about $S$
    \item \textbf{Flexible Threshold:} Can adjust $t$ based on privacy requirements
\end{itemize}
\end{frame}
